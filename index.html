# src/app/core/cleaning.py
def clean_dataframe(df: pd.DataFrame) -> pd.DataFrame:
    """Limpieza completa con funciones puras"""
    df = drop_fully_empty_columns(df)
    df = normalize_column_names(df)

    # Parseo de fechas españolas
    date_cols = ["fec_not", "fec_con", "ini_sin"]
    for c in date_cols:
        if c in df.columns:
            df[c] = df[c].apply(parse_spanish_date)

    # Conversiones numéricas
    df["edad"] = pd.to_numeric(df["edad"], errors="coerce")
    df["codigo_sspm"] = pd.to_numeric(df["codigo_sspm"], errors="coerce")

    return df

Parseo de Fechas Españolas

def parse_spanish_date(value: Any) -> Optional[date]:
    """Convierte '3-ene-2023' a datetime.date"""
    if not value: return None
    m = re.match(r"^(\d{1,2})-([a-zñ]{3})-(\d{4})$", str(value).lower())
    if not m: return None

    months = {"ene":1, "feb":2, "mar":3, "abr":4, "may":5, "jun":6,
              "jul":7, "ago":8, "sep":9, "oct":10, "nov":11, "dic":12}

    day, mon, year = int(m.group(1)), m.group(2), int(m.group(3))
    month = months.get(mon)
    if not month: return None

    try:
        return date(year, month, day)
    except ValueError:
        return None

Normalización de Nombres

def normalize_column_names(df: pd.DataFrame) -> pd.DataFrame:
    """Normaliza nombres: minúsculas, sin espacios, sin '_' finales"""
    new_cols = []
    for c in df.columns:
        c2 = c.strip().lower()
        c2 = re.sub(r"_+$", "", c2)  # quita underscores finales
        new_cols.append(c2)
    df_copy = df.copy()
    df_copy.columns = new_cols
    return df_copy
